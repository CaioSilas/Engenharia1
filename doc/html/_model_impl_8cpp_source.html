107void ModelImpl::run(int timeStart, int timeEnd, int timeVariance){
108
109 ModelImpl::SystemIterator systemIt;
110 ModelImpl::FlowIterator flowIt;
111
112 System *sourceValue, *destinyValue;
113
114 for(time = timeStart; time < timeEnd; time +=timeVariance){
115 for(flowIt = FlowBegin(); flowIt < FlowEnd(); flowIt++){
116 (*flowIt)->setValue((*flowIt)->execute());
117 }

119 for(flowIt = FlowBegin(); flowIt < FlowEnd(); flowIt++){

121 sourceValue->setValue(sourceValue->getValue() - (*flowIt)->getValue());
122
123 destinyValue = (*flowIt)->getDestiny();
124 destinyValue->setValue(destinyValue->getValue() + (*flowIt)->getValue());
